using UnityEngine;
using System.Collections;
using System.Collections.Generic;

public static class ListUtil 
{
    public static List Merge(this List src, List b)
    {
        int count = b.Count;
        for(int i = 0; i < count; i ++)
        {
            src.Add(b[i]);
        }

        return src;
    }
}
